// Get LastPlayer Chaal info and show the chaal, throwcoin animation and potcoin
            string GamePlayRecord = "https://kopotron.com/php_rest_threecards/api/GamePlay/GetGamePlay.php?RoomId=" + RoomId.text;
            UnityWebRequest gameplay = UnityWebRequest.Get(GamePlayRecord);
            yield return gameplay.SendWebRequest();
            if (gameplay.isNetworkError || gameplay.isHttpError)
            {
                Debug.LogError(gameplay.error);
                yield break;
            }
            GamePlayInfo playInfo = new GamePlayInfo();
            playInfo = GamePlayInfo.CreateFromJSON(gameplay.downloadHandler.text);
            if(Seat2.text==playInfo.LastPlayer)
            {
                if(playInfo.PlayType!="Packed")
                {
                    // Show throw of coin
                    txtThrowCoin2.text = playInfo.ChaalCoin.ToString();
                    //P2PotCoin.SetActive(false);
                    P2PotCoin.SetActive(true);
                    P2Coin = P2PotCoin.GetComponent<Animator>();
                    if (P2Coin != null)
                    {
                        P2Coin.SetBool("Throw", true);
                    }
                }
                
                // update potcoin
                txtPotCoin.text = playInfo.PotCoin.ToString();
                // Show what chaal Player Played
                ShowChaal2.SetActive(true);
                txtChaal2.text = playInfo.PlayType;
                chaal2.SetBool("Show", true);
            }
            if (Seat3.text == playInfo.LastPlayer)
